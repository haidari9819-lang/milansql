Quickstart

Connect in seconds

REST API
# Login
curl -X POST https://milansql.de/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"you","password":"pw"}'

# Query
curl -X POST https://milansql.de/query \
  -H "Authorization: Bearer TOKEN" \
  -d '{"sql":"SELECT * FROM products"}'
JavaScript SDK
import MilanSQL from 'milansql';

const db = new MilanSQL(
  'https://milansql.de', token
);

const { rows } = await db
  .from('products')
  .select('name, price')
  .eq('category', 'IT')
  .order('price', { ascending: false })
  .limit(10)
  .execute();
MySQL Client
-- Any MySQL-compatible client
mysql -h milansql.de -P 4407 \
  -u alice -p

CREATE TABLE products (
  id   INT PRIMARY KEY AUTO_INCREMENT,
  name TEXT,
  price INT
);

INSERT INTO products
  VALUES (NULL, 'Laptop', 1200);
Semantic Search (AI)
-- Store embeddings
CREATE TABLE docs (
  id      INT,
  content TEXT,
  vec     VECTOR(3)
);

-- Cosine similarity search
SELECT content,
  vec <-> '[1,0,0]' AS dist
FROM docs
ORDER BY dist LIMIT 5;
